\section{Prosedur kerja}

\begin{enumerate}[leftmargin=*]
  \item Pelajari sintaks comprehension dari contoh.
  \item Bandingkan dengan loop klasik untuk kasus yang sama.
  \item Tulis satu generator dengan \texttt{yield}; iterasikan dengan \texttt{for} atau \texttt{next()}.
  \item Refaktor potongan kode loop panjang menjadi gaya idiomatik (aktivitas).
\end{enumerate}

\section{List dan Dict Comprehension}

\begin{lstlisting}[caption=Comprehension]
squares = [x * x for x in range(10) if x % 2 == 0]
nama_ke_id = {n.lower(): i for i, n in enumerate(["Ali", "Budi", "Ali"])}
\end{lstlisting}

\section{Generator dengan yield}

\begin{lstlisting}[caption=Generator sederhana]
def countdown(n):
    while n >= 0:
        yield n
        n -= 1

for x in countdown(3):
    print(x)
\end{lstlisting}

\subsection{Kode: \texttt{comprehension.py}}
\begin{lstlisting}[language=Python,caption={\texttt{comprehension.py}}]
"""Contoh penggunaan list comprehension dan generator."""


def deret_genap(batas: int):
    """Generator yang menghasilkan bilangan genap hingga batas."""
    for i in range(batas + 1):
        if i % 2 == 0:
            yield i

def main() -> None:
    # List comprehension
    angka = [1, 2, 3, 4, 5]
    kuadrat = [x**2 for x in angka if x % 2 != 0]
    print(f"Kuadrat ganjil: {kuadrat}")
    
    # Generator
    genap = deret_genap(10)
    print("Deret genap 0-10:", list(genap))

if __name__ == "__main__":
    main()
\end{lstlisting}
